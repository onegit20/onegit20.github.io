sudo mkdir /data
sudo pvcreate /dev/vdb
sudo vgcreate vg01 /dev/vdb
sudo lvcreate -l 100%FREE -n lv01 vg01
sudo mkfs.xfs /dev/vg01/lv01
sudo mount /dev/vg01/lv01 /data
echo '/dev/mapper/vg01-lv01  /data  xfs  defaults 0 0' | sudo  tee -a /etc/fstab
