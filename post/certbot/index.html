sudo certbot certonly --manual \
  -d yanyong.cc -d *.yanyong.cc \
  --email info@yanyong.cc \
  --agree-tos \
  --no-eff-email \
  --preferred-challenges=dns \

  --manual-cleanup-hook \
    'sleep 30 && id=`curl -sS -X GET "https://api.cloudflare.com/client/v4/zones/2109e1497fb164ac32a8a56f493jd767/dns_records?name=_acme-challenge.yanyong.cc" -H "X-Auth-Email: info@yanyong.cc" -H "X-Auth-Key: b16b2371170e9i9t7aed5df44a69f3k0fd947" -H "Content-Type: application/json" \
    | grep -oP "(?<=result\":\[\{\"id\":\")\w+"` \
    && curl -sS -X DELETE "https://api.cloudflare.com/client/v4/zones/2109e1497fb164ac32a8a56f493jd767/dns_records/$id" -H "X-Auth-Email: info@yanyong.cc" -H "X-Auth-Key: b16b2371170e9i9t7aed5df44a69f3k0fd947" -H "Content-Type: application/json"' \
