#!/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin  #设置PATH环境变量


#彩色打印[yes]、[no]、[warning]
echo_yes() { echo -e "\e[32m  [yes]\e[0m"; }  #绿色
echo_no() { echo -e "\e[31m  [no]\e[0m"; }  #红色
echo_warning() { echo -e "\e[33m  [warning]\e[0m"; }  #黄色

#显示脚本用法
usage() {
echo "Usage: `basename $0` [-h] [-n hostname] [-p port] [-t timezone] [-c ntp]

Options:
    -h             : 显示帮助
    -n hostname    : 设置主机名
    -p port        : 设置ssh端口, 默认2208
    -t timezone    : 设置时区, 默认Asia/Shanghai(查看所有可用时区timedatectl list-timezones)
    -c ntp         : 设置同步的ntp服务器, 默认192.168.1.200
"
}

#获取传入参数
get_arguments() {
	while getopts hn:p:t:c: opt
	do
		case "$opt" in
			h) usage && exit 0;;
			n) set_hostname=${OPTARG};;
			p) set_port=${OPTARG};;
			t) set_tz=${OPTARG};;
			c) set_ntp=${OPTARG};;
			*) exit 1;;
		esac
	done
}

#检测当前用户是否可以sudo免密执行指令
#其实不用强制检测免密，能sudo就可以了，执行过程中输一次可以sudo的当前用户密码就可以了
check_sudo() {
	if sudo -n true &> /dev/null; then
		echo_yes
	else
		echo_no
		exit 1
	fi
}

#检测是否非root执行脚本
#用$USER变量检测不准确
#######################################################
#    $ whoami; echo $USER                             #
#    user1                                            #
#    user1                                            #
#    $ su root                                        #
#    Password:                                        #
#    # whoami; echo $USER                             #
#    root                                             #
#    user1                                            #
#######################################################
check_not-root() {
	if [[ "root" == `whoami` ]]; then
		echo_no
		exit 1
	else
		echo_yes
	fi
}

#配置主机名
set_hostname() {
	if [[ "X" != "X""$set_hostname" ]]; then  #如果传入hostname参数
		sudo hostnamectl set-hostname $set_hostname
		echo_yes
	else
		echo_warning
	fi
}

#配置zabbix-agent
set_zabbix-agent() {
	:  #略
}

#配置时区及时间同步
set_timezone_ntp() {
	#timezone
	set_tz="${set_tz:-Asia/Shanghai}"  #不传时区参数的话，默认设置Asia/Shanghai时区
	sudo timedatectl set-timezone $set_tz
	
	#ntp
	set_ntp="${set_ntp:-192.168.1.200}"  #不传ntp服务器参数的话，默认192.168.1.200
	sudo sed -ri 's/^#?(NTP=).*$/\1'"$set_ntp"'/' /etc/systemd/timesyncd.conf  #正则^#?表示不管NTP那一行是否有注释，直接替换
	sudo systemctl restart systemd-timesyncd
	echo_yes
}

#配置文件打开数及进程数
set_openfile_process() {
echo '* soft nofile 65535
* hard nofile 65535
* soft nproc 65535
* hard nproc 65535' | sudo tee /etc/security/limits.d/nofile-noproc.conf > /dev/null
echo_yes
}

#配置ssh端口
set_ssh-port() {
	#设置端口
	set_port="${set_port:-2208}"  #不传ssh端口参数的话，默认设置2208
	
	#检测当前ssh端口
	config_port=`awk '/^Port/{print $2}' /etc/ssh/sshd_config`
	ssh_port="${config_port:-22}"  #如果config_port为空则是22端口
	
	#如果设置端口与当前端口不一致，才执行
	if [[ "$set_port" != "$ssh_port" ]]; then
		if grep -q '^Port [0-9]\+' /etc/ssh/sshd_config; then  #如果配置文件有指定Port
			sudo sed -ri "s/^Port [0-9]+$/Port $set_port/" /etc/ssh/sshd_config
		else  #否则没有指定Port，比如默认22端口的情况
			echo "Port $set_port" | sudo tee -a /etc/ssh/sshd_config > /dev/null
		fi
		sudo systemctl reload ssh
	fi
	echo_yes
}


#主函数
main() {
	get_arguments "$@"
	echo -n " 1/7) 是否非root执行脚本"; check_not-root
	echo -n " 2/7) 是否可以sudo免密执行指令"; check_sudo
	echo -n " 3/7) 配置主机名"; set_hostname
	echo -n " 4/7) 配置时区及时间同步"; set_timezone_ntp
	echo -n " 5/7) 配置ssh端口"; set_ssh-port
	echo -n " 6/7) 配置文件打开数及进程数"; set_openfile_process
	echo -n " 7/7) 配置zabbix-agent"; set_zabbix-agent
}


#脚本执行入口
main "$@"
